47 uint32_t MaxMicroseconds = 2000>
48 static result<rc_servo> create(hal::pwm& p_pwm)
49 {
50 // Check that MinMicroseconds is less than MaxMicroseconds
51 static_assert(
52 MinMicroseconds < MaxMicroseconds,
53 "The minimum microseconds must be less than maximum microseconds.");
54 // Check that Frequency period is greater than MaxMicroseconds
55 static_assert(
56 (1000.0 / Frequency) > (MaxMicroseconds / 1000.0),
57 "The maximum microseconds is greater than the period of the frequency");
58 constexpr auto frequency = hertz{ Frequency };
59
60 // Check if any errors happened while setting the frequency of the pwm.
61 // Using HAL_CHECK will return any errors that occur from the factory
62 // function allowing the caller to choose what to do with the error
63 // information.
64 HAL_CHECK(p_pwm.frequency(frequency));
65
66 // Calculate the wavelength in microseconds.
67 auto wavelength = (1.0f / frequency) * std::micro::den;
68 // min_percent represents the minimum percentage to be used with the pwm
69 // signal. The percentage is calculated by using the minimum width of the
70 // signal in microseconds divided by the wavelength to get the decimal
71 // representation of the percentage.
72 auto min_percent = percentage(MinMicroseconds / wavelength);
73 // max_percent represents the maximum percentage to be used with the pwm
74 // signal. The percentage is calculated by using the maximum width of the
75 // signal in microseconds divided by the wavelength to get the decimal
76 // representation of the percentage.
77 auto max_percent = percentage(MaxMicroseconds / wavelength);
78 // percent_range holds float value of min_percent and max_percent for use
79 // with map function used in position().
80 auto percent_range =
81 std::make_pair(min_percent.value(), max_percent.value());
82 // If no errors happen, call the constructor with verified parameters
83 return rc_servo(p_pwm, percent_range);
84 }

103 //
104 // Example:
105 // pwm min percentage: percentage(0.05)
106 // pwm max percentage: percentage(0.25)

115 m_percent_range);

117 // Set the duty cycle of the pwm with the scaled percent.
