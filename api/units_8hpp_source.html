127[[nodiscard]] constexpr std::int64_t cycles_per(
128 hertz p_source,
129 hal::time_duration p_duration) noexcept
130{
131 // Full Equation:
132 // / ratio_num \_
133 // frequency_hz * |period| * | ----------- | = cycles
134 // \ ratio_den /
135 //
136 // std::chrono::nanoseconds::period::num == 1
137 // std::chrono::nanoseconds::period::den == 1,000,000
138
139 const auto denominator = decltype(p_duration)::period::den;
140 const auto float_count = static_cast<float>(p_duration.count());
141 const auto cycle_count = (float_count * p_source) / denominator;
142
143 return static_cast<std::int64_t>(cycle_count);
144}

192[[nodiscard]] inline result<std::chrono::nanoseconds> duration_from_cycles(
193 hertz p_source,
194 uint32_t p_cycles) noexcept
195{
196 // Full Equation (based on the equation in cycles_per()):
197 //
198 //
199 // / cycles * ratio_den \_
200 // |period| = | ---------------------------|
201 // \ frequency_hz * ratio_num /
202 //
203 constexpr auto ratio_den = std::chrono::nanoseconds::period::den;
204 constexpr auto ratio_num = std::chrono::nanoseconds::period::num;
205 constexpr auto int_min = std::numeric_limits<std::int64_t>::min();
206 constexpr auto int_max = std::numeric_limits<std::int64_t>::max();
207 constexpr auto float_int_min = static_cast<float>(int_min);
208 constexpr auto float_int_max = static_cast<float>(int_max);
209
210 const auto source = std::abs(p_source);
211 const auto float_cycles = static_cast<float>(p_cycles);
212 const auto nanoseconds = (float_cycles * ratio_den) / (source * ratio_num);
213
214 if (float_int_min <= nanoseconds && nanoseconds <= float_int_max) {
215 return std::chrono::nanoseconds(static_cast<std::int64_t>(nanoseconds));
216 }
217 return new_error(std::errc::result_out_of_range);
218}
