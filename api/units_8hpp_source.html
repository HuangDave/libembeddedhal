233[[nodiscard]] constexpr std::int64_t cycles_per(
234 hertz p_source,
235 hal::time_duration p_duration) noexcept
236{
237 // Full Equation:
238 // / ratio_num \_
239 // frequency_hz * |period| * | ----------- | = cycles
240 // \ ratio_den /
241 //
242 // std::chrono::nanoseconds::period::num == 1
243 // std::chrono::nanoseconds::period::den == 1,000,000
244
245 const auto denominator = decltype(p_duration)::period::den;
246 const auto float_count = static_cast<float>(p_duration.count());
247 const auto cycle_count = (float_count * p_source) / denominator;
248
249 return static_cast<std::int64_t>(cycle_count);
250}

260template<typename Period>
261constexpr std::chrono::duration<int64_t, Period> wavelength(hertz p_source)
262{
263 using duration = std::chrono::duration<int64_t, Period>;
264
265 static_assert(Period::num == 1, "The period ratio numerator must be 1");
266 static_assert(Period::den >= 1,
267 "The period ratio denominator must be 1 or greater than 1.");
268
269 constexpr auto denominator = static_cast<decltype(p_source)>(Period::den);
270 auto period = (1.0f / p_source) * denominator;
271
272 if (std::isinf(period)) {
273 return duration(std::numeric_limits<int64_t>::max());
274 }
275
276 return duration(static_cast<int64_t>(period));
277}

306[[nodiscard]] inline result<std::chrono::nanoseconds> duration_from_cycles(
307 hertz p_source,
308 uint32_t p_cycles) noexcept
309{
310 // Full Equation (based on the equation in cycles_per()):
311 //
312 //
313 // / cycles * ratio_den \_
314 // |period| = | ---------------------------|
315 // \ frequency_hz * ratio_num /
316 //
317 constexpr auto ratio_den = std::chrono::nanoseconds::period::den;
318 constexpr auto ratio_num = std::chrono::nanoseconds::period::num;
319 constexpr auto int_min = std::numeric_limits<std::int64_t>::min();
320 constexpr auto int_max = std::numeric_limits<std::int64_t>::max();
321 constexpr auto float_int_min = static_cast<float>(int_min);
322 constexpr auto float_int_max = static_cast<float>(int_max);
323
324 const auto source = std::abs(p_source);
325 const auto float_cycles = static_cast<float>(p_cycles);
326 const auto nanoseconds = (float_cycles * ratio_den) / (source * ratio_num);
327
328 if (float_int_min <= nanoseconds && nanoseconds <= float_int_max) {
329 return std::chrono::nanoseconds(static_cast<std::int64_t>(nanoseconds));
330 }
331 return new_error(std::errc::result_out_of_range);
332}
