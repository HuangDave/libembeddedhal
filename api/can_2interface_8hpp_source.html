124 [[nodiscard]] constexpr std::optional<std::uint32_t> is_valid(
125 hertz p_operating_frequency) const noexcept
126 {
127 // 1. propagation_delay, phase_segment1, phase_segment2 and
128 // synchronization_jump_width must be nonzero.
129 if (propagation_delay == 0 || phase_segment1 == 0 ||
130 phase_segment2 == 0 || synchronization_jump_width == 0) {
131 return std::nullopt;
132 }
133
134 // 2. synchronization_jump_width must be the lesser between phase_segment1
135 // and phase_segment2.
136 if (synchronization_jump_width > 4 ||
137 synchronization_jump_width > phase_segment1 ||
138 synchronization_jump_width > phase_segment2) {
139 return std::nullopt;
140 }
141
142 const std::uint16_t bit_width_v = bit_width();
143
144 // 3. The total bit width must be equal to or greater than 8 Tq/bit; the
145 // sum of sync_segment, propagation_delay, phase_segment1 and
146 // phase_segment2.
147 if (bit_width_v < 8) {
148 return std::nullopt;
149 }
150
151 // 4. The CAN device's operating frequency must be at least 8 times the
152 // bit rate to give the minimum.

154 // width must be close enough to an integer to produce a usable BRP.

157 const float baud_rate_prescaler = p_operating_frequency / scaled_baud;
158 const auto final_prescaler = std::lround(baud_rate_prescaler);
